Scheduler — Ishlarni navbatga qo'yish va boshqarish

Scheduler har bir update (setState, render) uchun priority belgilaydi va ularni min-heap da saqlaydi. Har browser frame da maksimal 5ms ishlaydi, keyin boshqaruvni browserga qaytaradi (time slicing). Bu animatsiya va input larni to'sib qo'ymaydi.

JS — Scheduler core Min-Heap + MessageChannel + Time Slicing 
// ════════════════════════════════════════════════════════════════
// 1. PRIORITY LEVELS
//    Har bir update qanchalik tez bajarilishi kerakligini belgilaydi.
//    IMMEDIATE → foydalanuvchi inputi, darhol.
//    IDLE      → prefetch, bo'sh vaqtda.
// ════════════════════════════════════════════════════════════════

const Priority = {
  IMMEDIATE: 1,  // onClick, onKeyDown — darhol (timeout = -1)
  HIGH:      2,  // hover, focus — 100ms ichida
  NORMAL:    3,  // setState — 500ms ichida
  LOW:       4,  // background update — 5000ms
  IDLE:      5,  // prefetch — cheksiz
};

// ════════════════════════════════════════════════════════════════
// 2. MIN-HEAP — O(log n) push/pop, Array.sort() dan 10x tezroq
//    Heap invariant: heap[parent] <= heap[child] (expirationTime bo'yicha)
//    heap[0] — har doim eng shoshilinch task
// ════════════════════════════════════════════════════════════════

const taskHeap = [];

function heapPush(heap, task) {
  heap.push(task);
  siftUp(heap, heap.length - 1);  // Yangi elementni to'g'ri joyga ko'tar
}

function heapPop(heap) {
  if (heap.length === 0) return null;
  const top  = heap[0];          // Eng muhim task
  const last = heap.pop();        // Oxirgi elementni olamiz
  if (heap.length > 0) {
    heap[0] = last;               // Oxirgini boshga qo'yamiz
    siftDown(heap, 0);            // Pastga "sho'ng'iting"
  }
  return top;
}

function siftUp(heap, i) {
  while (i > 0) {
    const parent = (i - 1) >> 1; // Parent indeks: floor((i-1)/2)
    if (heap[parent].expirationTime > heap[i].expirationTime) {
      [heap[parent], heap[i]] = [heap[i], heap[parent]]; // Swap
      i = parent;
    } else break;
  }
}

function siftDown(heap, i) {
  const n = heap.length;
  while (true) {
    let smallest = i;
    const l = 2 * i + 1, r = 2 * i + 2; // Chap va o'ng farzand
    if (l < n && heap[l].exp < heap[smallest].exp) smallest = l;
    if (r < n && heap[r].exp < heap[smallest].exp) smallest = r;
    if (smallest === i) break;       // To'g'ri joyda — to'xtat
    [heap[smallest], heap[i]] = [heap[i], heap[smallest]];
    i = smallest;
  }
}

// ════════════════════════════════════════════════════════════════
// 3. TIME SLICING — har frame da max 5ms ishlash
//    MessageChannel: setTimeout(0) dan tezroq (~1ms),
//    microtask dan sekinroq (browser paint ga to'sqinlik qilmaydi)
// ════════════════════════════════════════════════════════════════

const FRAME_YIELD_MS = 5; // 5ms = ~1/12 ta 60fps frame

function performWorkUntilDeadline() {
  const deadline  = performance.now() + FRAME_YIELD_MS;
  const shouldYield = () => performance.now() >= deadline;

  let task = heapPeek(taskHeap);
  while (task !== null) {
    if (shouldYield() && task.expirationTime > performance.now()) {
      break; // Vaqt tugadi — browserga qaytamiz
    }
    heapPop(taskHeap);
    const cont = task.callback(shouldYield);
    if (typeof cont === 'function') {
      task.callback = cont;  // Chunking: davomi bor
      heapPush(taskHeap, task);
    }
    task = heapPeek(taskHeap);
  }

  // Ish qolgan bo'lsa — keyingi frame da davom etamiz
  if (heapPeek(taskHeap) !== null) requestHostCallback();
}
Fiber Node — Virtual DOM dan boyroq ma'lumot strukturasi

Har bir React element (div, h1, MyComponent) uchun bitta Fiber node yaratiladi. Fiber oddiy Virtual DOM dan farqli — uni to'xtatish, davom ettirish, bekor qilish mumkin. child → sibling → return ko'rsatkichlari DFS traversal imkonini beradi.

JS — Fiber node strukturasi
// ════════════════════════════════════════════════════════════════
// FIBER NODE — har bir element/component uchun yaratiladi
//
// Tree navigatsiya (DFS):
//   child   — birinchi farzand
//   sibling — keyingi aka-uka
//   return  — ota (parent)
//
// Traversal:
//   A → A.child (B) → B.child (D) → D.sibling (E)
//   → E.return (B) → B.sibling (C) → C.return (A) → DONE
// ════════════════════════════════════════════════════════════════

function createFiber(type, props, key = null) {
  return {
    // ── Identifikatsiya ──────────────────────────────────────────
    type,      // "div" | "span" | MyComponent funksiyasi
    key,       // <li key="id"> — reconciler uchun
    props,     // { className, onClick, children: [...] }

    // ── Tree navigatsiya ─────────────────────────────────────────
    child:   null,  // Birinchi farzand fiberi
    sibling: null,  // Keyingi sibling fiberi
    return:  null,  // Parent fiberi (return = ota)
    index:   0,     // Siblings orasidagi tartib (diffing uchun)

    // ── DOM / Instance ───────────────────────────────────────────
    stateNode: null, // Haqiqiy DOM node | null (FC uchun)

    // ── Double buffering ─────────────────────────────────────────
    alternate: null, // current.alternate = wIP, wIP.alternate = current
                     // Commit bo'lgach ular rol almashtiradi

    // ── Hooks ────────────────────────────────────────────────────
    memoizedState: null, // Hook linked list boshi
                         // hook → hook.next → hook.next → null
    updateQueue: null,   // Pending setState action lar

    // ── Effect flags (bit mask) ──────────────────────────────────
    flags: NoFlags,       // Placement | Update | Deletion | LayoutEffect
    subtreeFlags: NoFlags, // Pastki daraxtdagi barcha flaglar yig'indisi
                            // (optimizatsiya: 0 bo'lsa subtree ni o'tkazib yub)

    // ── Effect list (completeWork quradi) ────────────────────────
    firstEffect: null, // Linear zanjir boshi
    lastEffect:  null, // Linear zanjir oxiri
    nextEffect:  null, // Zanjirdagi keyingi effect
                       // commitRoot shu zanjirni O(effects) da aylanadi

    ref: null, // useRef yoki ref prop
  };
}

// ── Effect flag lar (bit mask) ────────────────────────────────────
// Bir vaqtda bir necha flag bo'lishi mumkin: flags |= Placement | Ref
// ─────────────────────────────────────────────────────────────────
const NoFlags      = 0b000000; // Hech qanday amal kerak emas
const Placement    = 0b000001; // DOM ga qo'shish (appendChild/insertBefore)
const Update       = 0b000010; // DOM ni yangilash (props diff)
const Deletion     = 0b000100; // DOM dan o'chirish (removeChild)
const LayoutEffect = 0b001000; // useLayoutEffect ishga tushirish
const PassiveEffect= 0b010000; // useEffect ishga tushirish
const Ref          = 0b100000; // ref ni yangilash
Reconciler — Key-based Diffing (O(n) algoritm)

Yangi element array bilan eski fiber tree ni taqqoslaydi. type bir xil → UPDATE (DOM saqlanadi). type farq → PLACEMENT + eski DELETION. key bo'lsa — element harakatini to'g'ri kuzatadi.

JS — reconcileChildrenKey-Map + lastPlacedIndex optimization
// ════════════════════════════════════════════════════════════════
// reconcileChildren — DFS da beginWork chaqiradi
//
// Kirish:
//   wipFiber — hozir ishlanayotgan workInProgress fiberi
//   elements — yangi virtual DOM elementlar array si
//
// Chiqish:
//   wipFiber.child — yangi fiber tree boshlanmasi
//   deletions[]    — o'chiriladigan fiberlar
// ════════════════════════════════════════════════════════════════

function reconcileChildren(wipFiber, elements) {

  // ── 1. Eski children → Map yaratamiz ────────────────────────
  //    key mavjud bo'lsa key, bo'lmasa index ishlatamiz.
  //    Bu O(n) lookup imkonini beradi (Array.find O(n²) bo'ladi).
  // ──────────────────────────────────────────────────────────
  const oldFiberMap = new Map();
  let temp = wipFiber.alternate?.child;
  let idx  = 0;
  while (temp) {
    oldFiberMap.set(temp.key ?? temp.index, temp);
    temp = temp.sibling; idx++;
  }

  let prevSibling = null;
  let lastPlacedIndex = 0; // DOM harakatini minimizatsiya qilish uchun

  for (let i = 0; i < elements.length; i++) {
    const el  = elements[i];
    const key = el?.key ?? i;
    const old = oldFiberMap.get(key);

    // ── 2. Type tekshirish ───────────────────────────────────
    const sameType = old && el && old.type === el.type;

    if (sameType) {
      // ✅ UPDATE — xuddi shu DOM node ni qayta ishlatamiz
      //    stateNode (real DOM) va memoizedState (hooks) saqlanadi!
      newFiber = {
        ...old,
        props:     el.props,     // Yangi props
        alternate: old,          // Eski fiberni alternate ga
        flags:     Update,       // DOM props diff qilish kerak
      };
      oldFiberMap.delete(key); // Ishlatildi — delete

      // lastPlacedIndex: element o'z joyida qoldimi?
      // Agar old.index < lastPlacedIndex → DOM da pastroqda edi,
      // endi ustga chiqishi kerak → Placement (insertBefore)
      if (old.index < lastPlacedIndex) newFiber.flags |= Placement;
      else lastPlacedIndex = old.index;

    } else {
      // ❌ PLACEMENT — yangi DOM yaratamiz
      if (el) newFiber = { ...el, flags: Placement, alternate: null };
      // Eski fiber bor bo'lsa — o'chiramiz
      if (old) { old.flags = Deletion; deletions.push(old); }
    }

    // Fiber tree ga ulash
    if (i === 0) wipFiber.child = newFiber;
    else if (prevSibling) prevSibling.sibling = newFiber;
    prevSibling = newFiber;
  }

  // ── 3. Qolgan eski fiberlar → barini DELETION qilamiz ───────
  //    Bu yerda xato: eski kod faqat oxirgi index da tekshirgan!
  //    To'g'ri: loop dan tashqarida barcha qolganlarni o'chiramiz.
  oldFiberMap.forEach(fiber => {
    fiber.flags = Deletion;
    deletions.push(fiber);
  });
}
beginWork — Fiber ga "kirish": component render + reconcile

Fiber "ochiladi": type ga qarab component chaqiriladi, children reconcile qilinadi. Real React da bu funksiya 15+ branch ga ega (ClassComponent, Suspense, Context...). Biz asosiy 3 tani implement qilamiz: ROOT, FunctionComponent, HostComponent.

JS — beginWork
// ════════════════════════════════════════════════════════════════
// beginWork — fiber "ochish" bosqichi
//
// Kirish:  wipFiber — ishlayotgan workInProgress fiberi
// Chiqish: wipFiber.child — keyingi ishlanadigan fiber
//          null — bu fiber da child yo'q (completeWork ga o'tamiz)
// ════════════════════════════════════════════════════════════════

function beginWork(wipFiber) {

  if (typeof wipFiber.type === 'function') {
    // Function Component: hooks + render chaqiriladi
    updateFunctionComponent(wipFiber);

  } else if (wipFiber.type === 'ROOT') {
    // Root fiber: faqat children reconcile
    reconcileChildren(wipFiber, wipFiber.props.children);

  } else {
    // Host component: div, span, h1... — DOM yaratib children reconcile
    updateHostComponent(wipFiber);
  }

  return wipFiber.child; // Keyingi fiber (child ga tushamiz)
}

// ── Function Component ────────────────────────────────────────────
// Hooks engine ni sozlaydi va component funksiyasini chaqiradi.
// useState, useEffect shu yerda ishlaydi.
// ─────────────────────────────────────────────────────────────────
function updateFunctionComponent(wipFiber) {
  // Global hook cursors ni sozlaymiz
  currentlyRenderingFiber = wipFiber;
  workInProgressHook      = null;
  currentHook             = null;

  // Hooks linked list ni tozalaymiz (render jarayonida qayta qo'shiladi)
  wipFiber.memoizedState = null;

  // Component chaqiriladi — barcha hooks shu yerda ishlaydi:
  //   useState  → mountWorkInProgressHook() yoki updateWorkInProgressHook()
  //   useEffect → hook ga _pendingSetup saqlanadi
  //   useRef    → hook ga { current } saqlanadi
  let children;
  try {
    children = wipFiber.type(wipFiber.props); // ← Asosiy chaqiruv
  } finally {
    currentlyRenderingFiber = null; // Har doim tozalaymiz
  }

  // Natijani array ga aylantirib reconcile qilamiz
  const arr = children
    ? (Array.isArray(children) ? children.flat() : [children])
    : [];
  reconcileChildren(wipFiber, arr);
}

// ── Host Component ────────────────────────────────────────────────
// div, span, h1 va boshqa native elementlar.
// Birinchi render → DOM yaratadi, keyingi render → diff qiladi.
// ─────────────────────────────────────────────────────────────────
function updateHostComponent(wipFiber) {
  if (!wipFiber.stateNode) {
    wipFiber.stateNode = createDom(wipFiber); // Yangi DOM yaratish
  }
  const children = [wipFiber.props.children].flat().filter(Boolean);
  reconcileChildren(wipFiber, children);
}
completeWork — Fiber dan "chiqish": effect list qurish

completeWork fiber "yopilganda" chaqiriladi (child yo'q yoki hammasi tugadi). Ikkita ish qiladi: 1) subtreeFlags ni parent ga "bubble up" qiladi — bu commit da butun subtree ni skip qilish imkonini beradi. 2) Effect list zanjirini quradi — commitRoot faqat effecti bor fiberlarni O(effects) da aylanadi, O(n) emas.

JS — completeWorksubtreeFlags bubble-up + effect list
// ════════════════════════════════════════════════════════════════
// completeWork — fiber "yopish" bosqichi
//
// Effect list nima?
//   Butun tree ni commit paytida aylanish O(n).
//   completeWork linear zanjir quradi: faqat effecti borlar.
//   commitRoot shu zanjirni aylanadi → O(effects), n emas!
//
//   Root.firstEffect → fiberA → fiberB → fiberC → null
//                       (nextEffect bilan bog'liq)
// ════════════════════════════════════════════════════════════════

function completeWork(fiber) {
  const parent = fiber.return;
  if (!parent) return; // Root fiberi — parent yo'q

  // ── 1. subtreeFlags bubble-up ───────────────────────────────
  //    Farzand flaglarini otaga ko'taramiz.
  //    Commit da: if (fiber.subtreeFlags === NoFlags) → skip!
  //    Bu katta tree larda sezilarli optimizatsiya.
  // ──────────────────────────────────────────────────────────
  parent.subtreeFlags |= fiber.subtreeFlags; // Pastki daraxt
  parent.subtreeFlags |= fiber.flags;        // O'zining flaglari

  // ── 2. Effect list zanjiri ──────────────────────────────────
  //    Faqat haqiqiy effecti bor fiberlarni qo'shamiz.
  //    Zanjir: parent.firstEffect → ... → parent.lastEffect
  // ──────────────────────────────────────────────────────────

  // a) Agar bu fiberning o'zida effect bor bo'lsa:
  if (fiber.flags !== NoFlags) {
    if (parent.lastEffect !== null) {
      parent.lastEffect.nextEffect = fiber; // Zanjir oxiriga qo'shamiz
    } else {
      parent.firstEffect = fiber;           // Birinchi effect
    }
    parent.lastEffect = fiber;
  }

  // b) Bu fiberning subtree effect list ini parent ga ko'chiramiz
  if (fiber.firstEffect !== null) {
    if (parent.lastEffect !== null) {
      // Parent ning oxiriga ulayamiz
      parent.lastEffect.nextEffect = fiber.firstEffect;
    } else {
      parent.firstEffect = fiber.firstEffect;
    }
    parent.lastEffect = fiber.lastEffect;
  }
}

// ── Work loop da completeWork qanday chaqiriladi ─────────────────
// DFS: pastga tushamiz (beginWork), oxirga yetsak — chiqamiz (completeWork)
// ─────────────────────────────────────────────────────────────────

// child tugadi → completeWork → sibling bor? → u ga beginWork
//              → sibling yo'q → return (parent) → completeWork → ...

function workLoopConcurrent(shouldYield) {
  while (workInProgress !== null) {
    if (shouldYield()) return workLoopConcurrent; // Pause

    const next = beginWork(workInProgress);       // "Kirish"

    if (next !== null) {
      workInProgress = next;                        // Pastga
    } else {
      completeWork(workInProgress);                // "Chiqish"
      if (workInProgress.sibling) {
        workInProgress = workInProgress.sibling;   // Yoniga
      } else {
        // Otaga qaytib uni ham yopamiz (DFS backtrack)
        let c = workInProgress.return;
        while (c) {
          completeWork(c);
          if (c.sibling) { workInProgress = c.sibling; break; }
          c = c.return;
        }
        if (!c) workInProgress = null; // Root — tugadi!
      }
    }
  }
  return null; // workLoopConcurrent tugadi
}
Work Loop — DFS traversal, time-sliced
JS — render + rerender + scheduleRenderWork
// ════════════════════════════════════════════════════════════════
// render — ilovani birinchi marta DOM ga chiqaradi
// ════════════════════════════════════════════════════════════════
export function render(element, container) {
  // Root workInProgress fiber yaratamiz
  workInProgressRoot = {
    type: 'ROOT',
    props: { children: [element] },
    stateNode: container,  // DOM container
    alternate: currentRoot, // Eski tree (birinchi render da null)
    flags: NoFlags, subtreeFlags: NoFlags,
    firstEffect: null, lastEffect: null, nextEffect: null,
    child: null, sibling: null, return: null,
    memoizedState: null, updateQueue: [], index: 0,
  };
  deletions      = [];
  workInProgress = workInProgressRoot;
  scheduleRenderWork(Priority.NORMAL);
}

// ════════════════════════════════════════════════════════════════
// rerender — state o'zgarganida yangi tree qurib commit qiladi
//
// MUHIM: agar allaqachon workInProgressRoot bor bo'lsa —
// qaytib ketamiz. Bu race condition ni oldini oladi:
//   setState(1) → rerender boshlanadi
//   setState(2) → rerender yana chaqiriladi
//   ← ikkinchisi birinchisini bekor qiladi!
// ════════════════════════════════════════════════════════════════
export function rerender(priority = Priority.NORMAL) {
  if (!currentRoot) return;           // Hali render qilinmagan
  if (workInProgressRoot !== null) return; // Allaqachon ishlayapti

  workInProgressRoot = {
    type:      'ROOT',
    props:     currentRoot.props,      // Xuddi shu props (children = app)
    stateNode: currentRoot.stateNode,  // Xuddi shu DOM container
    alternate: currentRoot,            // ← Eski tree ga ishora (double buffer)
    flags: NoFlags, subtreeFlags: NoFlags,
    firstEffect: null, lastEffect: null, nextEffect: null,
    child: null, sibling: null, return: null,
    memoizedState: null, updateQueue: [], index: 0,
  };
  deletions      = [];
  workInProgress = workInProgressRoot;
  scheduleRenderWork(priority);
}

// ════════════════════════════════════════════════════════════════
// scheduleRenderWork — Scheduler ga render ishini topshiradi
//
// Chunking pattern:
//   - workLoopConcurrent null qaytarsa → tugadi, commitRoot
//   - funksiya qaytarsa → scheduler uni keyingi frame da davom ettiradi
// ════════════════════════════════════════════════════════════════
function scheduleRenderWork(priority) {
  scheduleTask(function renderTask(shouldYield) {
    const cont = workLoopConcurrent(shouldYield);

    if (cont !== null) {
      return renderTask; // Vaqt tugadi — davomini qaytaramiz (chunking!)
    }

    // Barcha fiberlar ishlandi — DOM ga yozamiz
    if (workInProgressRoot !== null) {
      commitRoot();
    }
    return null; // Tugadi
  }, priority);
}
Commit Phase — 3 sub-bosqich + Passive Effects

Commit to'xtatilmaydi — atom operatsiya. Effect list ni 3 marta aylanadi. before mutation → mutation (DOM) → layout (useLayoutEffect + refs). Undan keyin asinxron passive (useEffect).

JS — commitRoot va 3 sub-phase
// ════════════════════════════════════════════════════════════════
// commitRoot — barcha DOM o'zgarishlarini amalga oshiradi
//
// Nega 3 ta pass?
//   1. before: LayoutEffect cleanup (eski cleanup ni ishga tushirish)
//   2. mutation: DOM o'zgartirish (add/update/delete)
//   3. layout: useLayoutEffect setup + refs
//              Bu DOM o'zgarganidan keyin, paint OLDIDA bo'ladi.
//              Shuning uchun layout o'lchashlari to'g'ri.
// ════════════════════════════════════════════════════════════════

function commitRoot() {
  const finishedWork = workInProgressRoot;

  // Deletion lar effect list dan tashqarida — alohida boshqaramiz
  deletions.forEach(fiber => commitDeletion(fiber));
  deletions = [];

  // Effect list — completeWork qurib bergan linear zanjir
  // root.firstEffect → fiber1 → fiber2 → fiber3 → null
  let effect = finishedWork.firstEffect;

  // ── Pass 1: Before Mutation ─────────────────────────────────
  //    useLayoutEffect larning eski cleanup larini ishga tushuramiz.
  //    DOM hali o'zgarmagan!
  // ──────────────────────────────────────────────────────────
  let cursor = effect;
  while (cursor) {
    commitBeforeMutation(cursor);
    cursor = cursor.nextEffect;
  }

  // ── Pass 2: Mutation — haqiqiy DOM o'zgarishlari ────────────
  //    Placement: appendChild / insertBefore
  //    Update:    props diff → DOM yangilash
  //    Deletion:  removeChild (yuqorida bajarildi)
  // ──────────────────────────────────────────────────────────
  cursor = effect;
  while (cursor) {
    commitMutation(cursor);
    cursor = cursor.nextEffect;
  }

  // ── Double buffering: workInProgress → current ──────────────
  //    Bu nuqtadan boshlab currentRoot yangi tree.
  //    commitLayout da currentRoot dan foydalanish mumkin.
  // ──────────────────────────────────────────────────────────
  currentRoot        = workInProgressRoot;
  workInProgressRoot = null;

  // ── Pass 3: Layout — useLayoutEffect setup + refs ───────────
  //    DOM o'zgardi, lekin browser hali paint qilmadi.
  //    useLayoutEffect shu oynada: o'lchamlash, scroll o'rnatish.
  // ──────────────────────────────────────────────────────────
  cursor = effect;
  while (cursor) {
    commitLayout(cursor);
    cursor = cursor.nextEffect;
  }

  // ── Passive effects: useEffect (async) ──────────────────────
  //    Browser paint bo'lgandan keyin scheduler orqali.
  //    Foydalanuvchi ekran miltilashini ko'rmaydi.
  // ──────────────────────────────────────────────────────────
  if (rootWithPendingPassiveEffects) {
    scheduleTask(flushPassiveEffects, Priority.NORMAL);
  }
}

// ── commitPlacement: yangi fiber ni DOM ga qo'shish ──────────────
// insertBefore ishlatamiz — tartibni to'g'ri saqlash uchun
// ─────────────────────────────────────────────────────────────────
function commitPlacement(fiber) {
  if (!fiber.stateNode) return;
  const parent = getHostParentFiber(fiber).stateNode;
  const before = getHostSibling(fiber); // Keyingi DOM sibling
  before
    ? parent.insertBefore(fiber.stateNode, before)
    : parent.appendChild(fiber.stateNode);
}

// ── getHostSibling: function component larni o'tib ketib DOM sibling topish
// ─────────────────────────────────────────────────────────────────
function getHostSibling(fiber) {
  let node = fiber;
  siblings: while (true) {
    while (!node.sibling) {
      if (!node.return || node.return.type === 'ROOT') return null;
      node = node.return;
    }
    node = node.sibling;
    while (typeof node.type === 'function') { // FC ni o'tkazib yub
      if (node.flags & Placement || !node.child) continue siblings;
      node = node.child;
    }
    if (!(node.flags & Placement)) return node.stateNode;
  }
}
useState / useReducer — Hook linked list
JS — useState va useReducermount + update, linked list
// ════════════════════════════════════════════════════════════════
// HOOKS ENGINE
//
// Qoida: hooks har render da bir xil tartibda chaqirilishi shart.
// Shuning uchun linked list — index emas, tartib muhim.
//
// fiber.memoizedState: hook1 → hook2 → hook3 → null
//
// Mount (birinchi render):
//   mountWorkInProgressHook() — yangi hook yaratadi
//
// Update (re-render):
//   updateWorkInProgressHook() — eski hookni topadi (currentHook cursor)
// ════════════════════════════════════════════════════════════════

function mountWorkInProgressHook() {
  const hook = {
    state:           null,                 // Hozirgi qiymat
    queue:           { pending: null, dispatch: null }, // Update queue
    next:            null,                 // Keyingi hook
    _isLayoutEffect: false,
    _pendingSetup:   null,
  };
  if (!workInProgressHook) {
    currentlyRenderingFiber.memoizedState = hook; // Birinchi hook
  } else {
    workInProgressHook.next = hook;               // Linked list ga qo'shamiz
  }
  workInProgressHook = hook;
  return hook;
}

function updateWorkInProgressHook() {
  // currentHook cursor — eski fiber dagi mos hookni ko'rsatadi
  if (!currentHook) {
    currentHook = currentlyRenderingFiber.alternate?.memoizedState; // 1-hook
  } else {
    currentHook = currentHook.next; // Keyingi hook
  }
  // Eski hookdan state va queue ni ko'chiramiz
  const newHook = {
    state:  currentHook?.state ?? null,
    queue:  currentHook?.queue ?? { pending: null, dispatch: null },
    next:   null,
    _isLayoutEffect: currentHook?._isLayoutEffect ?? false,
    _pendingSetup:   null,
  };
  if (!workInProgressHook) {
    currentlyRenderingFiber.memoizedState = newHook;
  } else {
    workInProgressHook.next = newHook;
  }
  workInProgressHook = newHook;
  return newHook;
}

// ════════════════════════════════════════════════════════════════
// useReducer — useState ning generallashgan varianti
// useState = useReducer bilan (state, action) => action reducer
// ════════════════════════════════════════════════════════════════
export function useReducer(reducer, initialState) {
  const isMount = !currentlyRenderingFiber.alternate;
  const hook    = isMount ? mountWorkInProgressHook() : updateWorkInProgressHook();

  if (isMount) {
    // Birinchi render — state ni initialize qilamiz
    hook.state = typeof initialState === 'function'
      ? initialState()     // Lazy init: useState(() => heavyCalc())
      : initialState;

    // dispatch closure — hook va fiber ga bog'liq
    const dispatch = (action) => {
      hook.queue.pending = action; // Pending action saqlaymiz
      rerender(Priority.NORMAL);   // Yangi render schedule qilamiz
    };
    hook.queue.dispatch = dispatch;
    return [hook.state, dispatch];

  } else {
    // Re-render — pending action ni qo'llaymiz
    if (hook.queue.pending !== null) {
      // reducer(currentState, action) → newState
      hook.state = reducer(hook.state, hook.queue.pending);
      hook.queue.pending = null; // Tozalaymiz
    }
    return [hook.state, hook.queue.dispatch];
  }
}

// useState = useReducer shorthand
export function useState(init) {
  return useReducer(
    (state, action) => typeof action === 'function' ? action(state) : action,
    init
  );
}
useEffect / useLayoutEffect — Side effects va cleanup
JS — useEffectImpl + flushPassiveEffects
// ════════════════════════════════════════════════════════════════
// useEffect vs useLayoutEffect
//
//  useEffect      — asinxron, browser paint DAN KEYIN
//                   fetch, subscription, logger
//
//  useLayoutEffect — sinxron, DOM o'zgarganidan keyin, paint OLDIDA
//                   DOM o'lchamlash, scroll restore, tooltip pozitsiya
//
// Ikkalasi ham deps array bilan ishlaydi:
//   deps yo'q  → har render da
//   deps = []  → faqat mount/unmount da
//   deps = [x] → x o'zgarganda
// ════════════════════════════════════════════════════════════════

function useEffectImpl(callback, deps, isLayout) {
  const isMount = !currentlyRenderingFiber.alternate;
  const hook    = isMount ? mountWorkInProgressHook() : updateWorkInProgressHook();

  hook._isLayoutEffect = isLayout;

  const prevDeps    = hook.state?.deps;
  const prevCleanup = hook.state?.cleanup;

  // ── Deps taqqoslash ─────────────────────────────────────────
  //    Object.is — NaN === NaN va -0 !== 0 ni to'g'ri qiladi
  //    deps = undefined → har doim re-run
  //    deps = [] → faqat mount da (bo'sh array hech o'zgarmaydi)
  // ──────────────────────────────────────────────────────────
  const hasChanged = isMount
    ? true
    : deps === undefined
      ? true
      : !deps.every((d, i) => Object.is(d, prevDeps?.[i]));

  if (hasChanged) {
    // Setup funksiyasini commit bosqichi uchun saqlaymiz
    hook.state         = { deps, cleanup: prevCleanup };
    hook._pendingSetup = callback; // Commit da chaqiriladi

    // Fiber ga tegishli flag qo'yamiz
    currentlyRenderingFiber.flags |= isLayout ? LayoutEffect : PassiveEffect;

    // Eski cleanup ni unmount ro'yxatiga qo'shamiz
    if (!isMount && prevCleanup) {
      pendingPassiveEffects.unmount.push(currentlyRenderingFiber);
    }
  }
}

export const useEffect       = (cb, deps) => useEffectImpl(cb, deps, false);
export const useLayoutEffect = (cb, deps) => useEffectImpl(cb, deps, true);

// ════════════════════════════════════════════════════════════════
// flushPassiveEffects — useEffect ni ishga tushiradi (async)
//
// Tartib:
//   1. Avval barcha UNMOUNT cleanup lar (komponent ketdi)
//   2. Keyin barcha MOUNT setup lar (yangi effect)
//
// Nima uchun shu tartib?
//   Agar A → B o'zgarsa: A.cleanup() → B.setup()
//   Bu subscription managerlar uchun muhim.
// ════════════════════════════════════════════════════════════════

function flushPassiveEffects() {
  rootWithPendingPassiveEffects = null;

  // 1. Cleanup lar
  pendingPassiveEffects.unmount.forEach(fiber => {
    let hook = fiber.memoizedState;
    while (hook) {
      if (!hook._isLayoutEffect && hook.state?.cleanup) {
        hook.state.cleanup(); hook.state.cleanup = null;
      }
      hook = hook.next;
    }
  });

  // 2. Setup lar
  pendingPassiveEffects.mount.forEach(fiber => {
    let hook = fiber.memoizedState;
    while (hook) {
      if (!hook._isLayoutEffect && hook._pendingSetup) {
        const cleanup       = hook._pendingSetup();
        hook.state.cleanup  = typeof cleanup === 'function' ? cleanup : null;
        hook._pendingSetup  = null;
      }
      hook = hook.next;
    }
  });

  pendingPassiveEffects = { mount: [], unmount: [] };
  return null;
}

// ════════════════════════════════════════════════════════════════
// useRef — render lar o'rtasida saqlanuvchi mutable container
//
// Farqi useState dan:
//   useState — o'zgarsa re-render bo'ladi
//   useRef   — o'zgarsa re-render BO'LMAYDI
//
// Ishlatilishi:
//   - DOM ga murojaat: inputRef.current.focus()
//   - Timer id, WebSocket, setInterval ni saqlash
//   - Oldingi qiymatni eslab qolish (prevCount)
// ════════════════════════════════════════════════════════════════

export function useRef(initialValue) {
  const isMount = !currentlyRenderingFiber.alternate;

  if (isMount) {
    const hook = mountWorkInProgressHook();
    hook.state = { current: initialValue }; // Object — reference saqlanadi
    return hook.state; // Har doim bir xil { current } object qaytadi
  }
  // Re-render: xuddi shu object qaytadi (yangi yaratilmaydi!)
  return updateWorkInProgressHook().state;
}

// ════════════════════════════════════════════════════════════════
// useMemo — qimmat hisoblash natijasini kesh qiladi
//
// deps o'zgarmasa: cached qiymatni qaytaradi (factory chaqirilmaydi)
// deps o'zgansa:   factory qayta chaqiriladi, yangi qiymat saqlanadi
//
// Ishlatilishi:
//   const sorted = useMemo(() => bigArray.sort(), [bigArray]);
//   const total  = useMemo(() => items.reduce(...), [items]);
// ════════════════════════════════════════════════════════════════

export function useMemo(factory, deps) {
  const isMount = !currentlyRenderingFiber.alternate;

  if (isMount) {
    const hook  = mountWorkInProgressHook();
    const value = factory(); // Birinchi hisoblash
    hook.state  = { value, deps };
    return value;
  }

  const hook       = updateWorkInProgressHook();
  const prevDeps   = hook.state.deps;
  // Object.is — NaN, -0 ni to'g'ri taqqoslaydi
  const hasChanged = !deps.every((d, i) => Object.is(d, prevDeps[i]));

  if (hasChanged) {
    const value = factory(); // Qayta hisoblash
    hook.state  = { value, deps };
    return value;
  }
  return hook.state.value; // Kesh
}

// ════════════════════════════════════════════════════════════════
// useCallback — funksiyani memoize qiladi
//
// useCallback(fn, deps) = useMemo(() => fn, deps)
// Har render da yangi funksiya yaratilmaydi → child prop sifatida
// uzatilganda re-render oldini oladi (React.memo bilan birga).
// ════════════════════════════════════════════════════════════════

export function useCallback(callback, deps) {
  return useMemo(() => callback, deps);
}

// ════════════════════════════════════════════════════════════════
// createContext / useContext — global state prop drilling siz
//
// Sodda implementatsiya: context._currentValue ni to'g'ridan o'qiydi.
// To'liq: fiber tree da eng yaqin Provider ni topadi.
// ════════════════════════════════════════════════════════════════

export function createContext(defaultValue) {
  const context = {
    _currentValue: defaultValue,
    // Provider component — context qiymatini o'rnatadi
    Provider({ value, children }) {
      context._currentValue = value;
      return children;
    },
  };
  return context;
}

export function useContext(context) {
  // Hook linked list da saqlaymiz (hooks qoidasiga rioya)
  // To'liq: subscription yaratib, context o'zgaranda re-render
  const isMount = !currentlyRenderingFiber.alternate;
  isMount ? mountWorkInProgressHook() : updateWorkInProgressHook();
  return context._currentValue; // Hozirgi qiymat
}
